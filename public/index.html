const questions = require('./questions'); module.exports = async (req, res) => { const url = req.url || '/'; if (url === '/api/questions' && req.method === 'GET') { const safe = questions.map(q => { const { answer, workings, ...rest } = q; return rest; }); return res.json(safe); } if (url === '/api/check' && req.method === 'POST') { const { questionId, answer } = req.body; const q = questions.find(q => q.id === questionId); if (!q) return res.status(404).json({ error: 'Not found' }); let correct = false; if (q.type === 'mcq') { correct = parseInt(answer) === q.answer; } else if (q.type === 'multi') { const submitted = (Array.isArray(answer) ? answer : [answer]).map(Number).sort(); const expected = [...q.answer].sort(); correct = JSON.stringify(submitted) === JSON.stringify(expected); } else { const norm = s => String(s).toLowerCase().replace(/\s+/g, ' ').trim(); correct = norm(answer) === norm(q.answer); } return res.json({ correct, correctAnswer: q.answer, workings: q.workings || null, hint: q.hint || null }); } res.status(404).json({ error: 'Not found' }); };
